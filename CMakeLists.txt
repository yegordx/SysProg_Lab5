cmake_minimum_required(VERSION 3.26)
project(SysProg_Lab5)

set(CMAKE_CXX_STANDARD 20)

find_package(BISON)
find_package(FLEX)

bison_target(Parser
        src/parser.y
        ${CMAKE_CURRENT_BINARY_DIR}/Parser.cpp
        COMPILE_FLAGS "--debug"
        DEFINES_FILE ${CMAKE_CURRENT_BINARY_DIR}/Parser.hpp
        VERBOSE
)
flex_target(Scanner
        src/lexer.l
        ${CMAKE_CURRENT_BINARY_DIR}/Scanner.cpp
)
add_flex_bison_dependency(Scanner Parser)

add_executable(SysProg_Lab5
        src/Scanner.hpp
        ${BISON_Parser_OUTPUTS}
        ${FLEX_Scanner_OUTPUTS}
        src/main.cpp
        src/cli.cpp
        src/cli.hpp
        src/ast/TreeNode.cpp
        src/ast/TreeNode.hpp
        src/ast/TreeVisualizer.cpp
        src/ast/TreeVisualizer.hpp
)

# To make project files available from autogenerated files
target_include_directories(SysProg_Lab5 PRIVATE ${CMAKE_CURRENT_BINARY_DIR} src/)

target_link_libraries(SysProg_Lab5 ${FLEX_LIBRARIES} cgraph gvc)
